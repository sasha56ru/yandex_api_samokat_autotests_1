SELECT "Couriers".login,
        COUNT ("inDelivery")
        FROM "Couriers"
JOIN "Orders" ON "Orders"."courierId"="Couriers".id
WHERE "Orders"."inDelivery"=true
GROUP BY "Couriers".login;

SELECT track,
       CASE WHEN finished = true THEN 2
            WHEN cancelled = true THEN -1
            WHEN "inDelivery" = true THEN 1
            ELSE 0
            END AS status
FROM "Orders";
